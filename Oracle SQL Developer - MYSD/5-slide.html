-- 1. Eliminar datos (rápido, mantiene estructura)

TRUNCATE TABLE estudiantes;

 

-- 2. Eliminar tablas permanentemente (con dependencias)

DROP TABLE proyectos CASCADE CONSTRAINTS;

 

-- 3. Eliminar vistas asociadas

DROP VIEW vista_proyectos;

 

-- 4. Eliminar paquetes (procedimientos/funciones)

DROP PACKAGE utilidades_bd;

 

-- 5. ¡CRÍTICO! Liberar espacio real en disco (Papelera)

PURGE RECYCLEBIN;

 

-- 6. Confirmar transacciones pendientes

COMMIT;
